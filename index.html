// ============================================ // Abyss AI - بازی ماهی‌گیری (JavaScript) // ============================================ // ===== عناصر DOM ===== const scene = document.getElementById('scene'); const penguin = document.getElementById('penguin'); const fish = document.getElementById('fish'); const bobber = document.getElementById('bobber'); const exclamation = document.getElementById('exclamation'); const caughtFish = document.getElementById('caughtFish'); const fishingLine = document.getElementById('fishingLine'); const status = document.getElementById('status'); const fishCountEl = document.getElementById('fishCount'); const energyDisplay = document.getElementById('energyDisplay'); const energyBar = document.getElementById('energyBar'); const fishBtn = document.getElementById('fishBtn'); const feedBtn = document.getElementById('feedBtn'); // ===== وضعیت بازی ===== let fishCount = 0; let energy = 10; const maxEnergy = 10; let isFishing = false; let isWaiting = false; let fishVisible = false; let fishCaught = false; let fishX = 0, fishY = 0; let fishTimer = null; let clickTimer = null; // ===== به‌روزرسانی UI ===== function updateUI() { fishCountEl.textContent = fishCount; energyDisplay.textContent = energy; const pct = (energy / maxEnergy) * 100; energyBar.style.width = pct + '%'; if (pct > 50) energyBar.style.background = '#2ecc71'; else if (pct > 25) energyBar.style.background = '#f1c40f'; else energyBar.style.background = '#e74c3c'; } // ===== شروع ماهی‌گیری ===== function startFishing() { if (isFishing || isWaiting) { status.textContent = '⏳ صبر کن!'; status.style.color = '#f39c12'; return; } if (energy < 1) { status.textContent = '❌ انرژی کافی نیست!'; status.style.color = '#e74c3c'; return; } energy--; updateUI(); isFishing = true; status.textContent = '🎣 چوب به آب افتاد...'; status.style.color = '#3498db'; // نمایش چوب‌پران bobber.style.display = 'block'; bobber.style.left = (20 + Math.random() * 60) + '%'; bobber.style.top = (55 + Math.random() * 25) + '%'; // نمایش خط fishingLine.style.display = 'block'; fishingLine.style.left = '18%'; fishingLine.style.top = '35%'; fishingLine.style.height = '80px'; fishingLine.style.transform = 'rotate(15deg)'; // پنگوئن حرکت میکنه penguin.style.transform = 'rotate(-10deg)'; // زمان انتظار const waitTime = 3000 + Math.random() * 5000; fishTimer = setTimeout(showFish, waitTime); } // ===== نمایش ماهی ===== function showFish() { isFishing = false; isWaiting = true; fishVisible = true; // موقعیت ماهی fishX = 20 + Math.random() * 60; fishY = 58 + Math.random() * 25; fish.style.left = fishX + '%'; fish.style.top = fishY + '%'; fish.style.display = 'block'; // علامت تعجب exclamation.style.display = 'block'; exclamation.style.left = '10%'; exclamation.style.top = '15%'; // پنگوئن برگشت penguin.style.transform = 'rotate(0deg)'; status.textContent = '🐟 ماهی پیدا شد! کلیک کن! (۳ ثانیه)'; status.style.color = '#2ecc71'; // تایمر فرار clickTimer = setTimeout(fishEscaped, 3000); } // ===== گرفتن ماهی ===== function catchFish() { if (!isWaiting || !fishVisible) return; clearTimeout(clickTimer); isWaiting = false; fishVisible = false; fishCaught = true; fish.style.display = 'none'; exclamation.style.display = 'none'; bobber.style.display = 'none'; fishingLine.style.display = 'none'; // نمایش ماهی صید شده caughtFish.style.display = 'block'; caughtFish.style.animation = 'none'; void caughtFish.offsetWidth; caughtFish.style.animation = 'catchPop 0.5s ease forwards'; fishCount++; updateUI(); status.textContent = '🎉 ماهی گرفتی! +۱ 🐟'; status.style.color = '#2ecc71'; isFishing = false; // پنگوئن خوشحال penguin.classList.add('bounce'); setTimeout(() => penguin.classList.remove('bounce'), 300); setTimeout(() => { caughtFish.style.display = 'none'; }, 1500); } // ===== ماهی فرار کرد ===== function fishEscaped() { if (fishCaught) return; isWaiting = false; fishVisible = false; fish.style.display = 'none'; exclamation.style.display = 'none'; bobber.style.display = 'none'; fishingLine.style.display = 'none'; status.textContent = '😔 ماهی فرار کرد! دوباره امتحان کن.'; status.style.color = '#e67e22'; isFishing = false; } // ===== غذا دادن ===== function feedPenguin() { if (fishCount < 10) { status.textContent = `❌ به ۱۰ ماهی نیاز داری! (داری ${fishCount})`; status.style.color = '#e74c3c'; return; } if (energy >= maxEnergy) { status.textContent = '✅ انرژی کامل است!'; status.style.color = '#2ecc71'; return; } fishCount -= 10; energy = Math.min(energy + 5, maxEnergy); updateUI(); status.textContent = '🐧 پنگوئن خوشحال شد! +۵ انرژی ❤️'; status.style.color = '#2ecc71'; penguin.classList.add('bounce'); setTimeout(() => penguin.classList.remove('bounce'), 300); } // ===== کلیک روی صحنه ===== scene.addEventListener('click', function(e) { if (!isWaiting || !fishVisible) return; const rect = this.getBoundingClientRect(); const x = ((e.clientX - rect.left) / rect.width) * 100; const y = ((e.clientY - rect.top) / rect.height) * 100; const dist = Math.sqrt(Math.pow(x - fishX, 2) + Math.pow(y - fishY, 2)); if (dist < 12) { catchFish(); } else { status.textContent = '👆 نزدیک‌تر به ماهی کلیک کن!'; status.style.color = '#f39c12'; setTimeout(() => { if (isWaiting) { status.textContent = '🐟 ماهی پیدا شد! کلیک کن! (۳ ثانیه)'; status.style.color = '#2ecc71'; } }, 1000); } }); // ===== دکمه‌ها ===== fishBtn.addEventListener('click', startFishing); feedBtn.addEventListener('click', feedPenguin); // ===== کیبورد ===== document.addEventListener('keydown', function(e) { if (e.key === ' ' || e.key === 'Space') { e.preventDefault(); startFishing(); } if (e.key === 'f' || e.key === 'F') { feedPenguin(); } }); // ===== شروع ===== updateUI(); console.log('🐧 Abyss AI - بازی ماهی‌گیری راه‌اندازی شد!');
